Table product {
  prd_id integer [primary key]
  size integer
  price double
  
  //banner
  material_type integer
  //reproduction
  paper_type integer
  impression_type integer  
}

Table orders{
  odr_id integer [primary key]
  usr_id integer [primary key]
  prd_id integer [primary key]
  
  quantity integer
}

table sales{
  sale_id integer [primary key]
  odr_id integer 
  date date
  status integer

  value double
}

Table users {
  usr_id integer [primary key]
  email varchar
  name varchar
  password varchar
  admin integer
}

Ref: orders.usr_id > users.usr_id // many-to-one

Ref: "product"."prd_id" < "orders"."prd_id"

Ref: "sales"."odr_id" - "orders"."odr_id"
