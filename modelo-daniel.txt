drop database grafica;
create database grafica;
use grafica;

CREATE TABLE `produtos` (
  `id` integer auto_increment PRIMARY KEY,
  `nome` varchar(255),
  `estoque` integer,
  `preco` double,
  `imagem` blob
);

CREATE TABLE `servicos` (
  `id` integer auto_increment PRIMARY KEY,
  `nome` varchar(255),
  `descricao` varchar(255),
  `imagem` blob
);

CREATE TABLE `tipo` (
  `id` integer auto_increment PRIMARY KEY,
  `description` varchar(255),
  `servicos_id` integer
);

CREATE TABLE `tamanho_produto` (
  `id` integer auto_increment PRIMARY KEY,
  `nome` varchar(255),
  `product_id` integer
);

CREATE TABLE `tamanho_servicos` (
  `id` integer auto_increment PRIMARY KEY,
  `nome` varchar(255),
  `servicos_id` integer
);

CREATE TABLE `orders` (
  `id` integer auto_increment PRIMARY KEY,
  `user_id` integer,
  `address_id` integer,
  `date` timestamp
);

CREATE TABLE `users` (
  `id` integer auto_increment PRIMARY KEY,
  `username` varchar(255),
  `password` varchar(255),
  `email` varchar(255),
  `is_admin` Boolean,
  `created_at` timestamp
);

CREATE TABLE `order_products` (
  `id` integer auto_increment PRIMARY KEY,
  `produt_id` integer,
  `order_id` integer,
  `quantity` integer,
  `tipo` varchar(255),
  `tamanho` varchar(255),
  `value_on_sell` double
);

CREATE TABLE `pedido_servicos` (
  `id` integer auto_increment PRIMARY KEY,
  `servicos_id` integer,
  `orders_id` integer,
  `quantity` integer,
  `tipo` varchar(255),
  `tamanho` varchar(255),
  `valor_venda` double
);

CREATE TABLE `address` (
  `id` integer auto_increment PRIMARY KEY,
  `user_id` integer,
  `CEP` varchar(255),
  `Rua` varchar(255),
  `complemento` varchar(255),
  `bairro` varchar(255),
  `numero` integer
);

ALTER TABLE `address` ADD FOREIGN KEY (`id`) REFERENCES `users` (`id`);

ALTER TABLE `orders` ADD FOREIGN KEY (`id`) REFERENCES `users` (`id`);

ALTER TABLE `order_products` ADD FOREIGN KEY (`order_id`) REFERENCES `orders` (`id`);

ALTER TABLE `tipo` ADD FOREIGN KEY (`servicos_id`) REFERENCES `servicos` (`id`);

ALTER TABLE `tamanho_produto` ADD FOREIGN KEY (`product_id`) REFERENCES `produtos` (`id`);

ALTER TABLE `order_products` ADD FOREIGN KEY (`produt_id`) REFERENCES `produtos` (`id`);

ALTER TABLE `pedido_servicos` ADD FOREIGN KEY (`orders_id`) REFERENCES `orders` (`id`);

ALTER TABLE `pedido_servicos` ADD FOREIGN KEY (`servicos_id`) REFERENCES `servicos` (`id`);

ALTER TABLE `tamanho_servicos` ADD FOREIGN KEY (`servicos_id`) REFERENCES `servicos` (`id`);
