CREATE TABLE `produtos` (
  `id` integer PRIMARY KEY,
  `nome` varchar(255),
  `estoque` integer,
  `preco` double,
  `imagem` blob
);

CREATE TABLE `servicos` (
  `id` integer PRIMARY KEY,
  `nome` varchar(255),
  `descricao` varchar(255),
  `imagem` blob
);

CREATE TABLE `tipo` (
  `id` integer PRIMARY KEY,
  `description` varchar(255),
  `servicos_id` integer
);

CREATE TABLE `tamanho_produto` (
  `id` integer PRIMARY KEY,
  `nome` varchar(255),
  `product_id` integer
);

CREATE TABLE `tamanho_servicos` (
  `id` integer PRIMARY KEY,
  `nome` varchar(255),
  `servicos_id` integer
);

CREATE TABLE `orders` (
  `id` integer PRIMARY KEY,
  `user_id` integer,
  `address_id` integer,
  `date` timestamp
);

CREATE TABLE `users` (
  `id` integer PRIMARY KEY,
  `username` varchar(255),
  `password` varchar(255),
  `email` varchar(255),
  `created_at` timestamp
);

CREATE TABLE `order_products` (
  `id` integer PRIMARY KEY,
  `produt_id` integer,
  `order_id` integer,
  `quantity` integer,
  `tipo` varchar(255),
  `tamanho` varchar(255),
  `value_on_sell` double
);

CREATE TABLE `pedido_servicos` (
  `id` integer PRIMARY KEY,
  `servicos_id` integer,
  `orders_id` integer,
  `quantity` integer,
  `tipo` varchar(255),
  `tamanho` varchar(255),
  `valor_venda` double
);

CREATE TABLE `address` (
  `id` integer PRIMARY KEY,
  `user_id` integer,
  `CEP` varchar(255),
  `Rua` varchar(255),
  `complemento` varchar(255),
  `bairro` varchar(255),
  `numero` integer
);
